def extract(dataset,labels):
    hist = dict()
    total_len = len(dataset)
    for label in labels:
        hist[label] = 0
    for data in dataset:
        if data in labels:
            hist[data] += 1
    for key in hist.keys():
        hist[key] = (hist[key] / total_len) * 100
    return list(hist.values())

def pie(colName, title, under50):
    column = data[colName]
    dictionary = {}
    for i in range(len(column)):
        key = column[i]
        if key not in dictionary.keys():
          dictionary[key] = 0
        if under50:
            if label[i] == 0:
              dictionary[key] += 1
        else:
            dictionary[key] += label[i]
    
    dictionary = {k: v for k, v in sorted(dictionary.items(), key=lambda item: item[1])}
    del dictionary[' ?']
    x = np.array(list(dictionary.keys()))
    y = np.array(list(dictionary.values()))
    

    porcent = 100.*y/y.sum()
    
    colors = mcp.gen_color(cmap="bwr",n=10)

    patches, texts = plt.pie(y, startangle=90, radius=1.2, colors=colors)
    labels = ['{0} - {1:1.2f} %'.format(i,j) for i,j in zip(x, porcent)]

    sort_legend = True
    if sort_legend:
        patches, labels, dummy =  zip(*sorted(zip(patches, labels, y),
                                              key=lambda x: x[2],
                                              reverse=True))

    plt.legend(patches, labels, loc='center left', bbox_to_anchor=(-1, 1.),
               fontsize=8)
    plt.title(title)
    return plt
